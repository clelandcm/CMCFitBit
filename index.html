Merge Six Data Elements

df <- steps %>%
  left_join(distance) %>%
  left_join(floors) %>%
  left_join(minVery) %>%
  left_join(minSeden) %>%
  left_join(HRZPD) %>%
  left_join(RHR)

Intraday Data for a 30-day Period

mydays <- seq(as.Date("2017-05-01"), as.Date("2017-05-30"), 1)

intra.list <- vector("list", length(mydays))

for(i in seq_along(mydays)){
  intra.list[[i]] <- get_intraday_data(cookie, what = "heart-rate", as.character(mydays[i]))
}

intra.DF <- bind_rows(intra.list)

intra.DF %>%
  filter(!is.na(bpm) & bpm > 0) %>%
  summarize(n = n(),
            Mean = mean(bpm),

Plots

Miles by Date

ggplot(subset(df, steps > 0), aes(x=as.Date(time), y= distance)) + 
  geom_point(color = "blue", alpha = .33) + 
  ylab("Miles") + xlab("") + 
  stat_smooth(method='loess') +
  scale_y_continuous(breaks=seq(0,16,0.5)) +
  scale_x_date(breaks = date_breaks("2 months"), labels = date_format("%b-%Y")) +
  theme_minimal() + 
  theme(axis.text.x = element_text(angle = 45, hjust = 1),
        legend.position = "bottom")

Miles by Day of the Week

my.title <- paste("Fitbit January 1, 2014", "to", format(Sys.Date(), "%B %d, %Y"))

df$DAYOFWEEK <- as.factor(format(df$time, '%a'))
df$DAYOFWEEK <- ordered(df$DAYOFWEEK, levels=c('Sun','Mon','Tue','Wed','Thu','Fri','Sat'))

ggplot(subset(df, steps > 0), aes(DAYOFWEEK, distance)) + 
  geom_boxplot() + xlab("Day of the Week") + ylab("Miles Walked") + ggtitle(my.title) +
  scale_y_continuous(breaks=seq(0,16,0.5)) +
  theme_minimal()

Mileage Before and After NYU Meyers Move to First Avenue

df %>% 
  filter(steps > 0) %>%
  mutate(`After Move` = as.Date(time) > "2015-01-01") %>%
ggplot(aes(x=as.Date(time), y= distance, colour = `After Move`)) + 
  geom_point(size=2, shape=1) + ylab("Miles") + xlab("") + 
  stat_smooth(method='loess', se=FALSE) + 
  scale_y_continuous(breaks=seq(0,16,1)) +
  scale_x_date(breaks = date_breaks("2 months"), labels = date_format("%b-%Y")) +
  theme_minimal() + 
  theme(axis.text.x = element_text(angle = 45, hjust = 1),
        legend.position = "bottom")
